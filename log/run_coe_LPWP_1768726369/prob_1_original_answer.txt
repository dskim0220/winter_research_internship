```python
import gurobipy as gp
from gurobipy import GRB

def max_profit():
    """
    Solve the linear programming problem to maximize profit from producing color and black-and-white printers.
    """
    # Create a new model
    m = gp.Model("printer_production")
    
    # Define decision variables
    x = m.addVar(name="color_printers")  # Number of color printers
    y = m.addVar(name="bw_printers")     # Number of black and white printers
    
    # Set objective
    m.setObjective(200 * x + 70 * y, GRB.MAXIMIZE)
    
    # Add constraints
    m.addConstr(x <= 20, "color_team_constraint")
    m.addConstr(y <= 30, "bw_team_constraint")
    m.addConstr(x + y <= 35, "machine_constraint")
    
    # Optimize model
    m.optimize()
    
    # Return the optimal objective value
    if m.status == GRB.OPTIMAL:
        obj = int(m.objVal)
        print(f"Optimal solution found: {obj}")
        print(f"Number of color printers: {x.x}")
        print(f"Number of black and white printers: {y.x}")
    else:
        print("No optimal solution found.")

# Call the function to solve the problem
max_profit()
```

This code will output the optimal number of color and black-and-white printers to produce to maximize the company's profit, along with the corresponding profit value.
