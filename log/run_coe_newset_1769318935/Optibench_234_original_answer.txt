```python
from gurobipy import *

# Initialize the model
model = Model("logistics_distribution")

# Define decision variables
x = model.addVar(lb=10, ub=40, name="x")  # Trucks for GoodsX
y = model.addVar(lb=15, ub=35, name="y")  # Trucks for GoodsY
z = model.addVar(lb=0, ub=25, name="z")   # Trucks for GoodsZ
o_x = model.addVar(vtype=GRB.INTEGER, lb=0, name="o_x")  # Optimization investment for GoodsX
o_y = model.addVar(vtype=GRB.INTEGER, lb=0, name="o_y")  # Optimization investment for GoodsY
o_z = model.addVar(vtype=GRB.INTEGER, lb=0, name="o_z")  # Optimization investment for GoodsZ

# Objective function: Minimize total delivery cost
model.setObjective(1000*x + 1200*y + 1500*z + 10*o_x*(x + 100) + 12*o_y*(y + 100) + 15*o_z*(z + 100), GRB.MINIMIZE)

# Constraints
model.addConstr(x + y + z == 50, "Total_trucks")
model.addConstr(o_x + o_y + o_z <= 10, "Optimization_budget")
model.addConstr(15*o_z <= 3750, "Cost_reduction_GoodsZ")
model.addConstr(15*o_x + 12*o_y + 10*o_z >= 1000 - 10*x - 12*y - 15*z, "Combined_cost_reduction")
model.addConstr(o_x >= 0, "Non_negativity_o_x")
model.addConstr(o_y >= 0, "Non_negativity_o_y")
model.addConstr(o_z >= 0, "Non_negativity_o_z")

# Optimize the model
model.optimize()

# Print the results
if model.status == GRB.Status.OPTIMAL:
    print(f"Optimal Solution Found:")
    print(f"x = {x.X}")
    print(f"y = {y.X}")
    print(f"z = {z.X}")
    print(f"o_x = {o_x.X}")
    print(f"o_y = {o_y.X}")
    print(f"o_z = {o_z.X}")
    print(f"Total Delivery Cost = {model.objVal}")
else:
    print("No optimal solution found.")
```

This code sets up the Gurobi model to solve the logistics distribution problem as described. The decision variables, constraints, and objective function are correctly defined according to the problem statement. The model is then optimized, and the results are printed if an optimal solution is found.
